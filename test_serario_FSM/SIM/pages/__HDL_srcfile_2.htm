     4  module main(input clk,input rst,input in,output reg p,c);
     5  parameter NO_COIN=3'd0,RS1=3'd1,RS2=3'd2,PRODUCT=3'd3,CHANGE=3'd4;
     6  reg [2:0] state;
     7  
     8  always@(posedge clk,posedge rst) begin
     9  	if (rst) begin
     10  	 state<=NO_COIN;
     11  	 p<=0;
     12  	 c<=0;
     13  	end
     14  
     15  	else begin
     16  
     17  	case(state)
     18  		NO_COIN: begin
     19  		 state<=in? RS1 : RS2;
     20  		 p<=0;
     21  		 c<=0;
     22  		end
     23  
     24  		RS1 : begin
     25  		 state<=in ? RS2 : PRODUCT;
     26  		 p<=0;
     27  		 c<=0;
     28  		end
     29  
     30  		RS2 : begin
     31  		 state<= in ? PRODUCT : CHANGE;
     32  		 p<=0;
     33  		 c<=0;
     34  		end
     35  
     36  		PRODUCT : begin
     37  		 state <= NO_COIN;
     38  		 p<=1;
     39  		 c<=0;
     40  		end
     41  
     42  		CHANGE : begin
     43  		 state<=NO_COIN;
     44  		 p<=1;
     45  		 c<=1;
     46  		end
     47  
     48  		default : begin
     49  		 state<=NO_COIN;
     50  		end
     51  	
     52  	
     53  	endcase
     54      end
     55  
     56  end
     57  
     58  
     59  endmodule
